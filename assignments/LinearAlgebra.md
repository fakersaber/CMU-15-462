$$
\begin{split}
B(x,y) 
&= \frac{1}{2}(Q(x+y)-Q(x)-Q(y))\\
&= ax_1y_1 + bx_1y_2 + cx_2y_1 + dx_2y_2\\
&= 2(x_1+y_1)^2+2(x_1+y_1)(x_2+y_2)+8(x_2+y_2)^2 - 2x_1^2 - 2x_1x_2 - 8x_2^2 - 2y_1^2 - 2y_1y_2 - 8y_2^2\\
&= 4x_1y_1 + 2x_1y_2 + 2y_1x_2 + 16x_2y_2
\end{split}
$$

b= 2\\
c = 2\\
